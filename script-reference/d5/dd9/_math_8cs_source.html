37  /// </summary>
38  /// <returns>A boxed numeric object whos type is unsigned.</returns>
39  /// <remarks>This allows for converting without knowing the original type.</remarks>
40  /// <param name="value">The value.</param>
41  public static object ToUnsigned(object value)
42  {
43  switch (System.Type.GetTypeCode(value.GetType()))
44  {
45  case System.TypeCode.SByte:
46  return (byte)((sbyte)value);
47  case System.TypeCode.Int16:
48  return (ushort)((short)value);
49  case System.TypeCode.Int32:
50  return (uint)((int)value);
51  case System.TypeCode.Int64:
52  return (ulong)((long)value);
53 
54  case System.TypeCode.Byte:
55  return value;
56  case System.TypeCode.UInt16:
